interface Operation {
  id: string;
  type: 'insert' | 'delete' | 'update' | 'transform';
  blockId: string;
  data: any;
  timestamp: number;
  batchId?: string;  // 批量操作 ID
}

class HistoryService {
  private undoStack: Operation[] = [];
  private redoStack: Operation[] = [];
  private maxStackSize: number = 100;
  private currentBatchId: string | null = null;

  constructor(private editor: LexicalEditor);

  /** 记录操作 */
  record(operation: Operation): void;

  /** 开始批量操作 */
  startBatch(): string;

  /** 结束批量操作 */
  endBatch(): void;

  /** 撤销 */
  undo(): boolean;

  /** 重做 */
  redo(): boolean;

  /** 清空历史 */
  clear(): void;

  /** 是否可撤销 */
  canUndo(): boolean;

  /** 是否可重做 */
  canRedo(): boolean;
}

interface InputRule {
  // 规则名称
  name: string;

  // 触发文本（支持多个）
  triggers: string[];

  // 规则类型
  type: 'inline' | 'block';

  // 正则匹配（可选）
  pattern?: RegExp;

  // 转换函数
  transform: (match: RegExpMatchArray, text: string) => void;

  // 优先级（数字越大优先级越高）
  priority?: number;
}

class InputRuleService {
  private rules: InputRule[] = [];

  /** 注册规则 */
  register(rule: InputRule): void;

  /** 处理输入 */
  handleInput(text: string, position: number): boolean;

  /** 清空规则 */
  clear(): void;
}

// === 内置规则示例 ===
const headingRule: InputRule = {
  name: 'heading',
  triggers: ['# '],
  type: 'block',
  pattern: /^(#{1,6})\s(.*)$/,
  transform: (match, text) => {
    const level = match[1].length;
    const content = match[2];
    // 将当前段落转换为对应级别的标题
    editor.transformToHeading(level, content);
  },
  priority: 10
};

const boldRule: InputRule = {
  name: 'bold',
  triggers: ['**', '__'],
  type: 'inline',
  pattern: /\*\*(.+?)\*\*/g,
  transform: (match, text) => {
    // 将 **text** 转换为加粗格式
    editor.applyFormat('bold', match[1]);
  },
  priority: 5
};

const codeBlockRule: InputRule = {
  name: 'code-block',
  triggers: ['```'],
  type: 'block',
  transform: (match, text) => {
    // 创建代码块
    editor.insertBlock('code');
  },
  priority: 10
};
