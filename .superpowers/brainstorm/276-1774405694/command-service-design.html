CommandService 详细设计

命令系统 - 统一命令注册、执行、快捷键管理

1. 核心职责

📦 命令注册

管理所有编辑命令的注册表

⚡ 命令执行

执行命令、权限检查、状态反馈

⌨️ 快捷键绑定

快捷键 → 命令映射、冲突解决

🔘 工具栏同步

命令状态 → UI 按钮状态

📢 命令历史

记录执行的命令，支持撤销

🔌 插件扩展

支持第三方注册自定义命令

2. 接口定义

interface CommandConfig {
  /** 命令名称（唯一标识）*/
  name: string;

  /** 命令描述 */
  description?: string;

  /** 命令分类 */
  category: 'format' | 'insert' | 'delete' | 'transform';

  /** 执行函数 */
  execute: (editor: LexicalEditor, payload?: any) => void | Promise<void>;

  /** 是否可执行（权限检查）*/
  canExecute: (editor: LexicalEditor) => boolean;

  /** 是否激活（用于按钮高亮）*/
  isActive: (editor: LexicalEditor) => boolean;

  /** 快捷键绑定 */
  hotkey?: string | string[]; // 如："Mod+B", "Ctrl+Alt+1"

  /** 图标（用于工具栏/菜单）*/
  icon?: string;

  /** 标签（用于菜单显示）*/
  label?: string;
}

interface CommandRegistry {
  /** 注册命令 */
  register(config: CommandConfig): void;

  /** 注销命令 */
  unregister(name: string): void;

  /** 获取命令 */
  get(name: string): CommandConfig | undefined;

  /** 是否有命令 */
  has(name: string): boolean;

  /** 列出所有命令 */
  list(): CommandConfig[];

  /** 按分类筛选 */
  listByCategory(category: CommandConfig['category']): CommandConfig[];

  /** 执行命令 */
  execute(name: string, payload?: any): boolean;

  /** 检查是否可执行 */
  canExecute(name: string): boolean;

  /** 检查是否激活 */
  isActive(name: string): boolean;
}

4. 快捷键系统

// 快捷键绑定服务
class HotkeyService {
  private bindings: Map<string, string> = new Map(); // "Mod+B" -> "bold"

  /** 绑定快捷键 */
  bind(hotkey: string, commandName: string): void;

  /** 解绑快捷键 */
  unbind(hotkey: string): void;

  /** 解析快捷键字符串 */
  parse(hotkey: string): HotkeyConfig;

  /** 匹配键盘事件 */
  match(event: KeyboardEvent): string | null; // 返回命令名
}

// 快捷键修饰符
type Modifier = 'Ctrl' | 'Alt' | 'Shift' | 'Mod'; // Mod = Ctrl(Cmd on Mac)

// 快捷键格式示例
// "Mod+B"       → Ctrl+B (Windows) / Cmd+B (Mac)
// "Ctrl+Alt+1"  → Ctrl+Alt+1
// "Shift+Delete"→ Shift+Delete
