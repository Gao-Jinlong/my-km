2. EditorService 职责

📝 编辑器生命周期

创建、初始化、销毁 Lexical Editor 实例

🔄 文档操作

加载文档、保存文档、内容同步

✂️ 选区管理

追踪选区、光标位置、格式状态

📢 事件分发

向外部（AI 面板、文件树）发送事件

🔗 外部集成

与 AutoSave、AIContext 服务协作

⌨️ 命令执行

响应用户输入、执行编辑命令

3. 接口定义

interface IEditorService {
  // === 生命周期 ===

  /** 创建编辑器实例 */
  create(container: HTMLElement, options: EditorOptions): Promise<void>;

  /** 销毁编辑器实例 */
  destroy(): void;

  // === 文档操作 ===

  /** 加载文档 */
  loadDocument(doc: DocumentModel): Promise<void>;

  /** 保存文档 */
  saveDocument(): Promise<SaveResult>;

  /** 获取当前文档 */
  getCurrentDocument(): DocumentModel | null;

  // === 选区与状态 ===

  /** 获取选区内容 */
  getSelection(): SelectionState | null;

  /** 获取当前格式状态 */
  getFormatState(): FormatState;

  /** 是否可执行某命令 */
  canExecute(command: string): boolean;

  // === 命令执行 ===

  /** 执行编辑命令 */
  execute(command: string, payload?: any): boolean;

  // === 事件订阅 ===

  /** 订阅事件 */
  on<K extends keyof EditorEvents>(
    event: K,
    callback: EditorEvents[K]
  ): Disposable;

  // === 外部集成 ===

  /** 获取 AI 上下文 */
  getAIContext(): AIContext;

  /** 插入 AI 生成内容 */
  insertAIContent(content: string, options: InsertOptions): void;
}

4. EditorContainer 设计

// container/editor-container.ts
class EditorContainer {
  // === 单例 Registry ===
  private blockRegistry: BlockRegistry;
  private commandRegistry: CommandRegistry;

  // === 多实例管理 ===
  private editorServices: Map<string, EditorService> = new Map();
  private stores: Map<string, EditorStoreApi> = new Map();

  constructor() {
    // 初始化单例 Registry
    this.blockRegistry = new BlockRegistry();
    this.commandRegistry = new CommandRegistry();

    // 注册内置块类型
    this.registerBuiltinBlocks();
  }

  // === 工厂方法：创建编辑器实例 ===
  createInstance(documentId: string): EditorService {
    // 1. 创建 Zustand Store
    const store = createEditorStore(documentId);

    // 2. 创建依赖服务
    const autoSave = new AutoSaveService(
      store,
      this.blockRegistry
    );

    const aiContext = new AIContextService(
      store,
      this.blockRegistry
    );

    // 3. 创建 EditorService
    const editorService = new EditorService(
      documentId,
      store,
      autoSave,
      aiContext,
      this.blockRegistry,
      this.commandRegistry
    );

    // 4. 保存实例引用
    this.editorServices.set(documentId, editorService);
    this.stores.set(documentId, store);

    return editorService;
  }

  // === 获取实例 ===
  getInstance(documentId: string): EditorService | null {
    return this.editorServices.get(documentId) || null;
  }

  // === 销毁实例 ===
  disposeInstance(documentId: string): void {
    const service = this.editorServices.get(documentId);
    if (service) {
      service.destroy(); // 清理 Lexical Editor
      this.editorServices.delete(documentId);
      this.stores.delete(documentId);
    }
  }

  // === 获取 Store（用于 React 组件订阅）===
  getStore(documentId: string): EditorStoreApi | null {
    return this.stores.get(documentId) || null;
  }

  // === 内部方法：注册内置块类型 ===
  private registerBuiltinBlocks(): void {
    // 注册所有内置块类型
    this.blockRegistry.register(textBlockConfig);
    this.blockRegistry.register(headingBlockConfig);
    this.blockRegistry.register(codeBlockConfig);
    this.blockRegistry.register(tableBlockConfig);
    this.blockRegistry.register(imageBlockConfig);
    this.blockRegistry.register(equationBlockConfig);
    this.blockRegistry.register(listBlockConfig);
  }
}

// 全局容器实例
export const editorContainer = new EditorContainer();

6. 事件定义

interface EditorEvents {
  // === 文档事件 ===
  'document:loaded': (doc: DocumentModel) => void;
  'document:saved': (result: SaveResult) => void;
  'document:closed': () => void;

  // === 选区事件 ===
  'selection:changed': (selection: SelectionState) => void;
  'format:changed': (format: FormatState) => void;

  // === 内容事件 ===
  'content:changed': (content: string) => void;
  'content:dirty': () => void;
  'content:clean': () => void;

  // === AI 事件 ===
  'ai:request': (context: AIContext) => void;
  'ai:response': (response: AIResponse) => void;
  'ai:insert': (content: string) => void;

  // === 错误事件 ===
  'error': (error: EditorError) => void;
}
